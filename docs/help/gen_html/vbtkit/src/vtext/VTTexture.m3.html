 This module holds the textures for VTs. 


MODULE VTTexture;

IMPORT Pixmap;

VAR
  initflag: BOOLEAN;

PROCEDURE Init () RAISES {} =
  BEGIN
    IF  NOT initflag THEN
      gray := Pixmap.Gray;
      lightGray := gray;
      turn[FALSE] := Pixmap.Empty;
      turn[TRUE] := gray;
      initflag := TRUE;
    END;
  END Init;

BEGIN
  initflag := FALSE;
END VTTexture.
