INTERFACE Auth;

IMPORT Pathname, Thread;

TYPE
  T = RECORD
    server: TEXT;
    client: TEXT;
    password: TEXT;
    comment: TEXT;
  END;

  DB <: REFANY;

EXCEPTION Error(TEXT);

PROCEDURE Open(path: Pathname.T): DB
  RAISES {Error};

PROCEDURE Lookup(db: DB;
                 server: TEXT;
		 VAR (*OUT*) auth: T): BOOLEAN
  RAISES {Error, Thread.Alerted};

PROCEDURE Close(db: DB)
  RAISES {Error, Thread.Alerted};

END Auth.
