INTERFACE TextPortLogger;

IMPORT Logger, TextPort;

TYPE
  T <: Public;

  Public = Logger.T OBJECT METHODS
    init(tp: TextPort.T;
	 level: Logger.Priority := Logger.Priority.Debug): T;
  END;

 A TextPortLogger.T is a subtype of Logger.T that displays its
   output in a TextPort.  The tp parameter to init specifies
   the TextPort.T that is to be used to display the results. 


END TextPortLogger.
