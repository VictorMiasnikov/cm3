INTERFACE Version;

VAR
  Name: TEXT;  (* CONST *)

 Contains the CVS/RCS symbolic tag under which the sources were checked
   out, if any, or a string of the form U_yyyy_mm_dd_hh_mm_ss, where
   the lower-case letters represent the compilation date and time in GMT. 

  Target: TEXT;  (* CONST *)

 Contains the Modula-3 name of the platform for which the program was
   compiled. 

END Version.
