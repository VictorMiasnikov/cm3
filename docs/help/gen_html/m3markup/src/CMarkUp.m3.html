MODULE CMarkUp;

IMPORT Buf, Wx;

PROCEDURE Annotate (buf: Buf.T;  wx: Wx.T) =
  BEGIN
    Wx.PutText (wx, "<PRE>\n");
    Wx.PutStr  (wx, buf^);
    Wx.PutText (wx, "/n</PRE>\n");
  END Annotate;

BEGIN
END CMarkUp.
