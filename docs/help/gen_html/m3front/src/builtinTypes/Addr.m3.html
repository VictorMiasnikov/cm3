MODULE Addr;

IMPORT RefType, Tipe, Brand;

PROCEDURE Initialize () =
  BEGIN
    T := RefType.New (NIL, FALSE, Brand.New ("$address$"));
    Tipe.Define ("ADDRESS", T, TRUE);
  END Initialize;

BEGIN
END Addr.
