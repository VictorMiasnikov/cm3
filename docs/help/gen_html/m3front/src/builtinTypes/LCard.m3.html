MODULE LCard;

IMPORT SubrangeType, Target, TInt, Tipe, LInt;

PROCEDURE Initialize () =
  BEGIN
    T := SubrangeType.New (TInt.Zero, Target.Longint.max, LInt.T, TRUE);
    Tipe.Define ("LONGCARD", T, TRUE);
  END Initialize;

BEGIN
END LCard.
