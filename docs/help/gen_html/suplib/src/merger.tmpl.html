%
%  Quake functions to instantiate a generic merger.
%

readonly proc Merger (nm, elt) is
  local merger = nm & "Merger"
  build_generic_intf (merger, "Merger", [elt], VISIBLE)
  build_generic_impl (merger, "Merger", [elt])
end

readonly proc merger (nm, elt) is
  local merger = nm & "Merger"
  build_generic_intf (merger, "Merger", [elt], HIDDEN)
  build_generic_impl (merger, "Merger", [elt])
end
