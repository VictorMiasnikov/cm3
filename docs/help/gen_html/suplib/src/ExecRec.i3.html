INTERFACE ExecRec;

IMPORT GlobTree;

CONST
  Brand = "ExecRec";

TYPE
  T = OBJECT
    pattern: GlobTree.T;
    command: TEXT;
  END;

 An ExecRec.T describes a command to be executed when certain files
   are updated.  The pattern specifies which files the record applies
   to.  The command is a text string that will be passed to the command
   shell. 


END ExecRec.
