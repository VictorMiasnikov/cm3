INTERFACE RsyncBlock;

IMPORT MD5Digest, Word;

CONST Brand = "RsyncBlock";

TYPE
  T = RECORD
    num: CARDINAL;    (* Block number, starting with 0. *)
    rsum: Word.T;
    md5: MD5Digest.T;
  END;

PROCEDURE Compare(READONLY a, b: T): [-1..1];

 Compares by rsum and md5, but ignores num. 

PROCEDURE Equal(READONLY a, b: T): BOOLEAN;

 Compares by rsum and md5, but ignores num. 

PROCEDURE Hash(READONLY a: T): Word.T;

 Returns a.rsum. 

END RsyncBlock.
