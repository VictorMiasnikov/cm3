INTERFACE AuthMD5;

IMPORT IP;

PROCEDURE GenChallenge(peerAddr: IP.Address;
		       privateKey: TEXT): TEXT;

 Generates a random challenge using the given peer address and private
   key to stir it up. 

PROCEDURE GenResponse(challenge: TEXT;
                      sharedSecret: TEXT): TEXT;

 Generates a response to a challenge using the given shared secret. 

PROCEDURE CheckResponse(response: TEXT;
		        challenge: TEXT;
		        sharedSecret: TEXT): BOOLEAN;

 Equivalent to Text.Equal(response, GenResponse(challenge, sharedSecret)) 

PROCEDURE MakeSecret(realm: TEXT;
                     user: TEXT;
		     password: TEXT): TEXT;

 Scrambles together the given information to produce a shared secret
   suitable for GenResponse and CheckResponse.  This scrambled secret
   should be stored in the servers password database. 

END AuthMD5.
