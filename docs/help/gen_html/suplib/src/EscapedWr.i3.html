 The EscapedWr interface provides a subclass of Wr.T which
   supports record-oriented writes to a stream.  All text is passed on
   to the child Wr.T unchanged, except that a '.' at the beginning of
   a line is doubled.  When the EscapedWr.T is closed, an end of
   record indication is written to the child.  The end of record is
   .\n if it follows a newline in the text, or \n.+\n otherwise. 


INTERFACE EscapedWr;

IMPORT Wr;

TYPE
  T <: Public;

  Public = Wr.T OBJECT METHODS
    init(wr: Wr.T;
	 closeChild: BOOLEAN := TRUE): T;
  END;

   (* If "closeChild" is "TRUE" (the default), then closing the escaped
   writer will also cause the child writer to be closed.  If it is "FALSE",
   then the child writer will be left open. *)

END EscapedWr.
