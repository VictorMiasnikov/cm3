INTERFACE SysLogger;

IMPORT Logger;

EXCEPTION Error(TEXT);

TYPE
  T <: Public;

  Public = Logger.T OBJECT METHODS
    init(ident: TEXT;
	 facility: Facility;
	 options := Options{};
	 level: Logger.Priority := Logger.Priority.Debug): T;
  END;

  Facility = {
    Kern,
    User,
    Mail,
    Daemon,
    Auth,
    Syslog,
    Lpr,
    News,
    Uucp,
    Cron,
    Authpriv,
    Ftp,
    Local0,
    Local1,
    Local2,
    Local3,
    Local4,
    Local5,
    Local6,
    Local7
  };

  Option = {
    Pid,
    Cons,
    Ndelay,
    Nowait,
    Perror
  };

  Options = SET OF Option;

PROCEDURE DecodeFacility(name: TEXT): Facility
  RAISES {Error};

 Converts a textual name (e.g., uucp) to its corresponding Facility.
   The name may contain any combination of upper and lower case. 

PROCEDURE EncodeFacility(facility: Facility): TEXT;

 Converts a Facility to its corresponding name, e.g., Authpriv. 

END SysLogger.
