 Glob provides shell-style pattern matching for filenames. 


INTERFACE Glob;

TYPE
  MatchOption = {
    NoEscape,	(* Disable backslash escaping. *)
    Pathname,	(* Slash must be matched by slash. *)
    Period,	(* Leading period must be matched by period. *)
    (* Danger: The following are non-POSIX. *)
    LeadingDir, (* OK if pattern matches a leading path prefix. *)
    CaseFold,   (* Ignore case. *)
    PrefixDirs  (* Directory prefixes of pattern match too. *)
  };

  MatchOptions = SET OF MatchOption;

PROCEDURE Match(pattern, string: TEXT;
                options := MatchOptions{}): BOOLEAN;

 Return TRUE if the filename given by string matches the pattern
   given by pattern. 

END Glob.
