INTERFACE WatchDog;

IMPORT Thread, Time;

TYPE
  T <: ROOT;

PROCEDURE New(timeout: Time.T;
              target: Thread.T := NIL): T;

 Returns a new watchdog, set to alert target after timeout expires. 
 The target parameter defaults to the calling thread. 


PROCEDURE Reset(wd: T;
                timeout: Time.T := -1.0d0);

 Resets the timeout interval of the watchdog timer wd. 
 If timeout is unspecified, the most recent value is used again. 


PROCEDURE Cancel(wd: T);

 Cancels the watchdog timer wd.  The watchdog timer may not be used
   after this is called. 

PROCEDURE Expired(wd: T): BOOLEAN;

 Returns TRUE if the watchdog timer has expired and alerted the target
   thread. 

END WatchDog.
