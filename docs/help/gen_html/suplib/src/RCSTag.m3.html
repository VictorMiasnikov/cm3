MODULE RCSTag;

IMPORT RCSRevNum, Text;

PROCEDURE Compare(READONLY t1, t2: T): [-1..1] =
  BEGIN
    RETURN Text.Compare(t1.name, t2.name);
  END Compare;

PROCEDURE Equal(READONLY t1, t2: T): BOOLEAN =
  BEGIN
    RETURN Text.Equal(t1.name, t2.name);
  END Equal;

PROCEDURE IsCVSBranch(READONLY t: T): BOOLEAN =
  BEGIN
    RETURN RCSRevNum.IsCVSBranch(t^.revNum);
  END IsCVSBranch;

BEGIN
END RCSTag.
