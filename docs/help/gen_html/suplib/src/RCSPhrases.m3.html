MODULE RCSPhrases;

IMPORT RCSPhrase, RefSeq;

REVEAL
  T = RefSeq.T BRANDED OBJECT END;

  Iterator = IterPublic BRANDED OBJECT
    phrases: T;
    cur: CARDINAL;
  OVERRIDES
    next := IterNext;
  END;

PROCEDURE New(): T =
  BEGIN
    RETURN NEW(T).init();
  END New;

PROCEDURE Append(phrases: T; phrase: RCSPhrase.T) =
  BEGIN
    phrases.addhi(phrase);
  END Append;

PROCEDURE Iterate(phrases: T): Iterator =
  BEGIN
    RETURN NEW(Iterator, phrases := phrases, cur := 0);
  END Iterate;

PROCEDURE IterNext(self: Iterator;
                   VAR phrase: RCSPhrase.T): BOOLEAN =
  BEGIN
    IF self.phrases # NIL AND self.cur < self.phrases.size() THEN
      phrase := self.phrases.get(self.cur);
      INC(self.cur);
      RETURN TRUE;
    END;
    RETURN FALSE;
  END IterNext;

BEGIN
END RCSPhrases.
