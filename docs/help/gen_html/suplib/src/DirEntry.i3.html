 A DirEntry.T represents the information that is returned from reading
   a filesystem directory.  It is split out into a separate interface so
   that it can be used to instantiate generics. 


INTERFACE DirEntry;

IMPORT FileAttr;

CONST Brand = "DirEntry";

TYPE
  T = REF RECORD
    name: TEXT;
    attr: FileAttr.T;
  END;

PROCEDURE Compare(READONLY a, b: T): [-1..1];

 Returns Text.Compare(a.name, b.name). 

PROCEDURE Equal(READONLY a, b: T): BOOLEAN;

 Returns Text.Equal(a.name, b.name). 

END DirEntry.
