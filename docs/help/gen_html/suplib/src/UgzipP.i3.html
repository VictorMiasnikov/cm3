 This interface declares some external procedures that are private
   to the Ugzip implementation. 


INTERFACE UgzipP;

IMPORT Ugzip;

FROM Ctypes IMPORT const_char_star, int;

<*EXTERNAL*>
PROCEDURE deflateInit_(strm: Ugzip.z_stream_star;
                       level: int;
		       version: const_char_star;
		       stream_size: int): int;

<*EXTERNAL*>
PROCEDURE inflateInit_(strm: Ugzip.z_stream_star;
		       version: const_char_star;
		       stream_size: int): int;

END UgzipP.
