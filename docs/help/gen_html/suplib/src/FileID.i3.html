INTERFACE FileID;

IMPORT FileAttr, Word;

CONST
  Brand = "FileID";

TYPE
  T <: REFANY;

PROCEDURE FromAttr(fa: FileAttr.T): T;

 Returns the ID of the file with the given attributes.  If the necessary
   information isn't present in the attributes, returns NIL. 

PROCEDURE Equal(a, b: T): BOOLEAN;

PROCEDURE Hash(fid: T): Word.T;

END FileID.
