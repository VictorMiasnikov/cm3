INTERFACE Config;

IMPORT Point;

TYPE
  T = OBJECT
    name       : TEXT;
    nRows      : CARDINAL;
    nCols      : CARDINAL;
    nPieces    : CARDINAL;
    nTiles     : CARDINAL;
    delay      : LONGREAL;
    pieces     : PieceList;
  END;

CONST
  NRotations = 4;

TYPE
  PieceList = REF ARRAY (* 0..nPieces-1 *) OF Piece;
  Piece     = ARRAY [0..NRotations-1] OF RotatedPieceMap;
  TileList  = REF ARRAY (* 0..nTiles-1 *) OF Point.T;

  RotatedPieceMap = RECORD
    tiles   : TileList;
    hoffset : INTEGER;
    voffset : INTEGER;
  END;

PROCEDURE New (nSquares: [2..5]): T;

END Config.
