MODULE EmbProxiedObj;

REVEAL
  Proxy = PublicProxy BRANDED "EmbProxiedObj v1.0" OBJECT OVERRIDES
    init := Init;
    changeProxy := ChangeProxy;
  END;

PROCEDURE Init (self: Proxy; obj: REFANY): Proxy =
  BEGIN
    self.obj := obj;
    RETURN self;
  END Init;

PROCEDURE ChangeProxy (self: Proxy; obj: REFANY) =
  BEGIN
    self.obj := obj;
  END ChangeProxy;

BEGIN
END EmbProxiedObj.
