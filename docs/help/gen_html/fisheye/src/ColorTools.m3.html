MODULE ColorTools;

IMPORT Color, PaintOp;

PROCEDURE FromHSV (h, s, v: REAL;
                   mode            := PaintOp.Mode.Normal;
                   gray            := -1.0;
                   bw              := PaintOp.BW.UseIntensity):
  PaintOp.T =
  BEGIN
    WITH hsv = Color.HSV{h, s, v},
         rgb = Color.FromHSV(hsv)  DO
      RETURN PaintOp.FromRGB(rgb.r, rgb.g, rgb.b, mode, gray, bw)
    END
  END FromHSV;

BEGIN
END ColorTools.
