MODULE TextPortButton;

IMPORT VText;

PROCEDURE GetExtents(b: T; VAR (*out*) left, right: CARDINAL) =
  VAR
    l, r: VText.Index;
    unused1: VText.IntervalOptions;
    unused2: VText.OnOffState;
  BEGIN
    VText.ExplodeInterval(b.interval, l, r, unused1, unused2);
    left := l;  right := r;
  END GetExtents;

BEGIN
END TextPortButton.
