UNSAFE MODULE MachineIDPosix EXPORTS MachineID;
IMPORT MachineIDPosixC;

EXCEPTION Failure;
PROCEDURE Get (): T =
  <*FATAL Failure*>
  VAR id: T;
  BEGIN
    IF CanGet (id)
      THEN RETURN id;
      ELSE RAISE Failure;
    END;
  END Get;

PROCEDURE CanGet (VAR(*OUT*) id: T): BOOLEAN =
  BEGIN
    RETURN MachineIDPosixC.CanGet(ADR(id.r[0]));
  END CanGet;

BEGIN
END MachineIDPosix.
