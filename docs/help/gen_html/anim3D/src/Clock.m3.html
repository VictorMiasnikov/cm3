MODULE Clock;

FROM Time IMPORT Now;

REVEAL
  T = Public BRANDED OBJECT
  OVERRIDES
    init := Init;
    time := Time;
  END;

PROCEDURE Init (self : T) : T =
  BEGIN
    RETURN self;
  END Init;

PROCEDURE Time (<* UNUSED *> self : T) : LONGREAL =
  BEGIN
    RETURN Now () - clockstart;
  END Time;

VAR
  clockstart := Now ();

BEGIN
END Clock.
