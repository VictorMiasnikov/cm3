---------------------------------------------------------------------------

INTERFACE EnvUtils;

IMPORT TextTextTbl, Rd;

EXCEPTION Error(TEXT);

---------------------------------------------------------------------------

CONST Separators = SET OF CHAR{' ', '\t', '\r', '\n'};

---------------------------------------------------------------------------

TYPE T = TextTextTbl.T;

---------------------------------------------------------------------------

PROCEDURE ToText(env : T; sepChar := '\t') : TEXT RAISES {Error};

---------------------------------------------------------------------------

PROCEDURE FromRd(rd : Rd.T; sepChars := Separators; skipStart := FALSE) : T
  RAISES {Error};

---------------------------------------------------------------------------

PROCEDURE FromText(t : TEXT; sepChar := Separators) : T RAISES {Error};

END EnvUtils.
