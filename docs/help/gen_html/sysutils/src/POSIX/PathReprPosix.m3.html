MODULE PathReprPosix EXPORTS PathRepr;

IMPORT TextUtils;

--------------------------------------------------------------------------

PROCEDURE Native(pn : TEXT) : TEXT =
  BEGIN
    RETURN TextUtils.SubstChar(pn, '\\', '/');
    (* Unconditinal substitution actually performs better than this:
    IF Text.FindChar(pn, '\\') > -1  THEN
      RETURN TextUtils.SubstChar(pn, '\\', '/');
    ELSE
      RETURN pn;
    END;
    *)
  END Native;

BEGIN
  PathSepChar := '/';
  PathSep     := "/";
END PathReprPosix.
