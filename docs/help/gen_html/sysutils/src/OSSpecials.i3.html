INTERFACE OSSpecials;

 This module contains implementations of miscellaneous functionality that
   need to be different on POSIX, WindowsNT, OS/2, and DOS.
 


---------------------------------------------------------------------------

VAR
  LineBreak : TEXT; (* CONST *)

---------------------------------------------------------------------------

PROCEDURE QuotedProcessArgument(arg : TEXT) : TEXT;
  (* The brain damaged implementation of Process.Create on Windows NT
     does not preserve the boundaries between different arguments to
     a program. So if one argument contains one or more blanks (or other
     white space), it gets broken up into several arguments. This
     procedure is a workaround for this problem, not the correct
     solution...
   *)

END OSSpecials.
