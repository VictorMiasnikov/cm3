INTERFACE MFAlgs;

IMPORT MFGraph, R2;

TYPE
  MFEdge = MFGraph.Edge BRANDED OBJECT
      capacity, flow : REAL;
  END;

  MFVertex = MFGraph.Vertex BRANDED OBJECT
      wherefrom : MFEdge;
      marked : BOOLEAN;
      pos: R2.T; (* the GraphVBT position *)
      label: TEXT; (* the GraphVBT label *)
  END;

PROCEDURE NullText(g: MFGraph.T) : TEXT;

PROCEDURE PrintText(t: TEXT) : TEXT;

END MFAlgs.
