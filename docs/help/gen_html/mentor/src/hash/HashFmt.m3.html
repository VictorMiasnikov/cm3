UNSAFE MODULE HashFmt;

IMPORT Fmt;
IMPORT FormsVBT AS FV;
IMPORT IntList AS IL;

PROCEDURE FormsVBT (<*UNUSED*> form : FV.T) : TEXT =
  BEGIN
    RETURN "<a FormsVBT.T>";
  END FormsVBT;

PROCEDURE IntList (l : IL.T) : TEXT =
  VAR
    t := "";
  BEGIN
    WHILE l # NIL DO
      t := t & Fmt.Int (l.head);
      l := l.tail;
      IF l # NIL THEN
        t := t & ",";
      END;
    END;
    RETURN "[" & t & "]";
  END IntList;

BEGIN
END HashFmt.
