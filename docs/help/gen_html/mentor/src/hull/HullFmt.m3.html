MODULE HullFmt;

IMPORT Fmt;
IMPORT IntList AS IL;
IMPORT SiteList AS SL;

PROCEDURE IntList (l : IL.T) : TEXT =
  VAR
    t := "";
  BEGIN
    WHILE l # NIL DO
      t := t & Fmt.Int (l.head);
      l := l.tail;
      IF l # NIL THEN
        t := t & ",";
      END;
    END;
    RETURN "[" & t & "]";
  END IntList;

PROCEDURE SiteList (<* UNUSED *> l : SL.T) : TEXT =
  BEGIN
    RETURN "<a SiteList.T>";
  END SiteList;

BEGIN
END HullFmt.
