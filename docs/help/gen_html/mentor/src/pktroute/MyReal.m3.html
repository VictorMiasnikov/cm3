MODULE MyReal;

PROCEDURE Compare(a, b: T): [-1..1] =
  BEGIN
    IF a < b THEN RETURN -1
    ELSIF a > b THEN RETURN 1
    END;
    RETURN 0
  END Compare;

BEGIN
END MyReal.
