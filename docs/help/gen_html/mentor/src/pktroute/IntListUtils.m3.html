MODULE IntListUtils;

IMPORT IntList, Fmt;

PROCEDURE ToText(l: IntList.T): TEXT =
  VAR res := "["; BEGIN
    WHILE l # NIL DO
      res := res & Fmt.Int(l.head);
      l := l.tail;
      IF l # NIL THEN res := res & ", " END
    END;
    RETURN res & "]"
  END ToText;

BEGIN
END IntListUtils.
