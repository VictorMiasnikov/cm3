INTERFACE RefIntArray;

TYPE
  IntArray = ARRAY OF INTEGER;
  T = REF IntArray;

PROCEDURE ToText(ia: REF IntArray): TEXT;

 Return the elements of the array ia as a text of the form [ia[0], ia[1],
   ..., ia[LAST(ia)]. 

END RefIntArray.
