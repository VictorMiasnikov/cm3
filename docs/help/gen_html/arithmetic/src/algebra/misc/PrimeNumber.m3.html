PROCEDURE IsPrime (n: CARDINAL; ): BOOLEAN =
  (* is this number a prime number?*)
  (* may be replaced by the Miller-Rabin test or the very new test that
     need only polynomial time, but is still too slow for practical usage
     (today is 2003-05-11) *)
  VAR
    fac : CARDINAL;
    fac2: CARDINAL;              (* fac^2 *)
  BEGIN
    IF n < MaxPrime THEN RETURN primes[n] = 1; END;

    IF n MOD 2 = 0 THEN RETURN FALSE; END;

    fac := 3;
    fac2 := 9;
    WHILE fac2 <= n DO
      IF primes[fac] = 1 AND     (* this extra test is only useful if the
                                    modulo operation is very slow on the
                                    target machine *)
           n MOD fac = 0 THEN
        RETURN FALSE;
      END;
      INC(fac, 1);
      INC(fac2, 4 * fac);
      INC(fac, 1);
    END;
    RETURN TRUE;
  END IsPrime;

BEGIN
END PrimeNumber.
