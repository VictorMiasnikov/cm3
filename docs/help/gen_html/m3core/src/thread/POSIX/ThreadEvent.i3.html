INTERFACE ThreadEvent;

IMPORT ThreadF;

TYPE
  Kind = {Changed, Running, Deleted};

  T = RECORD
        kind: Kind;
        id: ThreadF.Id := 0;
        state := ThreadF.State.dead; END;

END ThreadEvent.
