MODULE WeakRef;

IMPORT RTWeakRef;

PROCEDURE FromRef (r: REFANY; p: CleanUpProc := NIL): T =
  BEGIN
    RETURN RTWeakRef.WeakRefFromRef(r, p);
  END FromRef;

PROCEDURE ToRef (w: T): REFANY =
  BEGIN
    RETURN RTWeakRef.WeakRefToRef(w);
  END ToRef;

BEGIN
END WeakRef.
