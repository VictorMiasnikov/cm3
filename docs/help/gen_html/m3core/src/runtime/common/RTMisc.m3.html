UNSAFE MODULE RTMisc;

------------------------------- rounded arithmetic ------------------------


PROCEDURE Upper (x, y: INTEGER): INTEGER =
  BEGIN
    RETURN ((x + y - 1) DIV y) * y;
  END Upper;

BEGIN
END RTMisc.
