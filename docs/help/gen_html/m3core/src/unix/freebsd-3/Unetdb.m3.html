MODULE Unetdb;

FROM Ctypes IMPORT int;

PROCEDURE VALID_SVC (a: int): BOOLEAN =
  BEGIN
    RETURN 0 < a AND a < NSVCS;
  END VALID_SVC;

BEGIN
END Unetdb.
